⚙ Setup Guide
3 minutes · free · no coding needed
1
Create a Google Sheet
Go to sheets.google.com → create a blank spreadsheet → name it HOLO TCG Tracker.
2
Open Apps Script and paste the code
In your sheet click Extensions → Apps Script. Delete everything and paste this code:
const CARDS_SHEET = 'Cards'; const SEALED_SHEET = 'Sealed'; const COLL_SHEET = 'Collection'; const PRICES_SHEET = 'Prices'; function doGet(e) { return respond({ status: 'ok', message: 'Holo Tracker script is running' }); } // Run this function ONCE manually in the script editor to grant UrlFetchApp permission function authorizeUrlFetch() { try { UrlFetchApp.fetch('https://api.tcgdex.net/v2/en/cards/sv01-1', {muteHttpExceptions:true}); Logger.log('UrlFetchApp authorized successfully!'); } catch(e) { Logger.log('Error: ' + e.message); } } function doPost(e) { try { const payload = JSON.parse(e.postData.contents); const ss = SpreadsheetApp.getActiveSpreadsheet(); let cs = ss.getSheetByName(CARDS_SHEET); if (!cs) { cs = ss.insertSheet(CARDS_SHEET); cs.appendRow(['id','name','set','game','notes','firstSeen','mentions','priceHistory']); } let sl = ss.getSheetByName(SEALED_SHEET); if (!sl) { sl = ss.insertSheet(SEALED_SHEET); sl.appendRow(['id','name','game','type','set','notes','firstSeen','lastSeen','mentions','lastPrice','priceHistory']); } let cl = ss.getSheetByName(COLL_SHEET); if (!cl) { cl = ss.insertSheet(COLL_SHEET); cl.appendRow(['id','cardName','set','game','condition','rawCondition','variation','tcgdexId','purchasePrice','purchaseCurrency','purchaseDate','notes','addedAt']); } let pr = ss.getSheetByName(PRICES_SHEET); if (!pr) { pr = ss.insertSheet(PRICES_SHEET); pr.appendRow(['cardId','tcgdexId','week','price','trend','avg7','avg30','low','fetchedAt','marketHistory']); } if (payload.action === 'getAll') { return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl), cmPrices: getAllCMPrices(pr) }); } if (payload.action === 'upsert') { upsertCard(cs, payload.card); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'delete') { deleteRow(cs, payload.id); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'upsertSealed') { upsertSealed(sl, payload.item); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'deleteSealed') { deleteRow(sl, payload.id); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'saveCMPrices') { saveCMPrices(pr, payload.rows); return respond({ status:'ok' }); } if (payload.action === 'upsertCollection') { upsertCollection(cl, payload.item); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'deleteCollection') { deleteRow(cl, payload.id); return respond({ cards: getAllCards(cs), sealed: getAllSealed(sl), collection: getAllCollection(cl) }); } if (payload.action === 'tcgdex') { try { const url = payload.url; if (!url || !url.startsWith('https://api.tcgdex.net/')) { return respond({ error: 'Invalid TCGdex URL' }); } const res = UrlFetchApp.fetch(url, { muteHttpExceptions: true }); const code = res.getResponseCode(); if (code !== 200) return respond({ error: 'TCGdex HTTP ' + code }); return respond({ data: JSON.parse(res.getContentText()) }); } catch(err) { return respond({ error: 'TCGdex fetch failed: ' + err.message }); } } return respond({ error: 'Unknown action' }); } catch(err) { return respond({ error: err.message }); } } function getAllCards(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return []; return rows.slice(1).map(r => ({ id: r[0], name: r[1], set: r[2], game: r[3], notes: r[4], firstSeen: r[5], mentions: r[6], priceHistory: JSON.parse(r[7] || '[]') })); } function getAllSealed(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return []; return rows.slice(1).map(r => ({ id: r[0], name: r[1], game: r[2], type: r[3], set: r[4], notes: r[5], firstSeen: r[6], lastSeen: r[7], mentions: r[8], lastPrice: r[9] || null, priceHistory: JSON.parse(r[10] || '[]') })); } function upsertCard(sheet, card) { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == card.id) { sheet.getRange(i+1,1,1,8).setValues([[ card.id, card.name, card.set, card.game||'', card.notes||'', card.firstSeen, card.mentions, JSON.stringify(card.priceHistory) ]]); return; } } sheet.appendRow([card.id, card.name, card.set, card.game||'', card.notes||'', card.firstSeen, card.mentions, JSON.stringify(card.priceHistory)]); } function upsertSealed(sheet, item) { const ph = JSON.stringify(item.priceHistory || []); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] == item.id) { sheet.getRange(i+1,1,1,11).setValues([[ item.id, item.name, item.game||'', item.type||'', item.set||'', item.notes||'', item.firstSeen, item.lastSeen, item.mentions, item.lastPrice||'', ph ]]); return; } } sheet.appendRow([item.id, item.name, item.game||'', item.type||'', item.set||'', item.notes||'', item.firstSeen, item.lastSeen, item.mentions, item.lastPrice||'', ph]); } function getAllCollection(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return []; return rows.slice(1).map(r => ({ id:r[0], cardName:r[1], set:r[2], game:r[3], condition:r[4], rawCondition:r[5]||null, variation:r[6]||null, tcgdexId:r[7]||null, purchasePrice:r[8]||null, purchaseCurrency:r[9]||'CHF', purchaseDate:r[10]||null, notes:r[11]||'', addedAt:r[12]||'' })); } function upsertCollection(sheet, item) { const rows = sheet.getDataRange().getValues(); const row = [item.id,item.cardName||'',item.set||'',item.game||'',item.condition||'raw',item.rawCondition||'',item.variation||'',item.tcgdexId||'',item.purchasePrice||'',item.purchaseCurrency||'CHF',item.purchaseDate||'',item.notes||'',item.addedAt||'']; for (let i = 1; i < rows.length; i++) { if (rows[i][0] == item.id) { sheet.getRange(i+1,1,1,13).setValues([row]); return; } } sheet.appendRow(row); } function getAllCMPrices(sheet) { const rows = sheet.getDataRange().getValues(); if (rows.length <= 1) return []; return rows.slice(1).map(r => [r[0],r[1],r[2],r[3],r[4],r[5],r[6],r[7],r[8],r[9]||'[]']); } function saveCMPrices(sheet, rows) { if (!rows || !rows.length) return; const existing = sheet.getDataRange().getValues(); const idx = {}; existing.slice(1).forEach((r,i) => { if(r[0]) idx[r[0]] = i+2; }); rows.forEach(row => { const cardId = row[0]; if(!cardId) return; if(idx[cardId]) { sheet.getRange(idx[cardId],1,1,9).setValues([row]); } else { sheet.appendRow(row); idx[cardId] = sheet.getLastRow(); } }); } function deleteRow(sheet, id) { const rows = sheet.getDataRange().getValues(); for (let i = rows.length - 1; i >= 1; i--) { if (rows[i][0] == id) { sheet.deleteRow(i + 1); return; } } } function respond(data) { return ContentService .createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); }
3
Deploy as a Web App
Click Deploy → New deployment → click the ⚙️ gear → select Web app
— Execute as: Me
— Who has access: Anyone
Click Deploy → when Google asks to authorize click Advanced → Go to project (unsafe) → Allow
Copy the URL ending in /exec
4
Paste URL and connect
Paste the URL into the Connect box at the top → click Connect. Done ✅
Next time you open this file it connects automatically.
